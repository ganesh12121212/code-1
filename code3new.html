heartCanvas.id = 'heartCanvas'; document.getElementById('heartParticles').appendChild(heartCanvas); const ctx = heartCanvas.getContext('2d'); let W, H; function resize() { W = window.innerWidth; H = window.innerHeight; heartCanvas.width = W; heartCanvas.height = H; } window.addEventListener('resize', resize); resize(); class Particle { constructor(){ this.x = Math.random()*W; this.y = Math.random()*H; this.size = Math.random()*8+4;
